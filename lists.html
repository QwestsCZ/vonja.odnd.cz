Lists in Lisp

List Basics

;; Creating lists
(list 1 2 3 4)
'(1 2 3 4)
(cons 1 (cons 2 nil))

;; Basic operations
(car '(1 2 3))     ; First element: 1
(cdr '(1 2 3))     ; Rest: (2 3)
(nth 1 '(1 2 3))   ; Second element: 2
(length '(1 2 3))  ; Length: 3

List Manipulation

;; Adding elements
(push 0 my-list)        ; Add to front
(append '(1 2) '(3 4))  ; Combine lists
(nconc list1 list2)     ; Destructive append

;; Removing elements
(remove 3 '(1 2 3 2 1)) ; Remove all 3s
(delete 3 my-list)      ; Destructive remove

List Processing

;; Mapping
(mapcar #'1+ '(1 2 3))     ; Add 1 to each
(maplist #'length '(a b c)) ; Length of each sublist

;; Filtering
(remove-if #'evenp '(1 2 3 4))    ; Remove evens
(remove-if-not #'numberp mixed-list) ; Keep numbers

Association Lists

;; Creating and using alists
(setq alist '((a . 1) (b . 2)))
(assoc 'a alist)         ; Find pair
(rassoc 2 alist)         ; Reverse lookup
(acons 'c 3 alist)       ; Add new pair
