Functions in Lisp

Function Definition

;; Basic function
(defun greet (name)
  (format t "Hello, ~a!" name))

;; Optional parameters
(defun power (base &optional (exponent 2))
  (expt base exponent))

;; Keyword parameters
(defun make-person (&key name (age 0) (city "Unknown"))
  (list name age city))

;; Rest parameters
(defun sum (&rest numbers)
  (apply #'+ numbers))

Lambda Functions

;; Anonymous function
(lambda (x) (* x x))

;; Using lambda with mapcar
(mapcar (lambda (x) (* x 2)) '(1 2 3 4))

Higher-Order Functions

;; Function as parameter
(defun apply-twice (f x)
  (funcall f (funcall f x)))

;; Using apply
(apply #'+ '(1 2 3 4))  ; Same as (+ 1 2 3 4)

;; Using funcall
(funcall #'+ 1 2 3 4)

Recursion

;; Factorial implementation
(defun factorial (n)
  (if (<= n 1)
      1
      (* n (factorial (- n 1)))))
